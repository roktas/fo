* atamaları

* işleciyle atama yap

    a, b, c = *('A'..'C')
    a #=> 'A'
    b #=> 'B'
    c #=> 'C'

    match, text, number = *"Something 981".match(/([A-z]*) ([0-9]*)/)
    match  #=> "Something 981"
    text   #=> "Something"
    number #=> "981"

case deyimlerinde Proc

case deyimlerinde Proc kullanılabilir (Ruby 1.9’da Proc#=== ile Proc#call eşdeğerdir)

    def multiple_of(factor)
      Proc.new { |product| product.modulo(factor).zero? }
    end

    case number
    when multiple_of(3)
      puts "Multiple of 3"
    when multiple_of(7)
      puts "Multiple of 7"
    end
